	changeentityproperty(P3, "defaultname", "ryopain");
	}
	if (P1 != NULL())

	killentity(P2);
	killentity(P3);
	changeopenborvariant("nojoin", 1);
	}
	else if (P2 != NULL())
	{
	killentity(P1);
	killentity(P3);
	changeopenborvariant("nojoin", 1);
	}
	else if (P3 != NULL())
	{
	killentity(P1);
	killentity(P2);
	changeopenborvariant("nojoin", 1);
	}
	if (anim1 == openborconstant("ANI_SPAWN") || anim1 == openborconstant("ANI_RESPAWN"))
	{
	changeentityproperty(P1, "health", 150);
	}
	else if (anim2 == openborconstant("ANI_SPAWN") || anim2 == openborconstant("ANI_RESPAWN"))
	{
	changeentityproperty(P2, "health", 150);
	}
	else if (anim3 == openborconstant("ANI_SPAWN") || anim3 == openborconstant("ANI_RESPAWN"))
	{
	changeentityproperty(P3, "health", 150);
	}
}
