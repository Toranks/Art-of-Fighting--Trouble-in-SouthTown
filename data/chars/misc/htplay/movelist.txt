facing   1
lifespan 5500
subject_to_gravity 0
setlayer	999999998

animationscript data/scripts/grabscript.c

palette	none




script	@script
void main()
{
	void self = getlocalvar("self");
	int plr = getentityproperty(self,"playerindex");
	int alt = getentityproperty(self,"a");
	int aggro = getentityproperty(self,"aggression");


	if ( playerkeys(plr,0, "movedown") && alt < aggro )
	{
		changeentityproperty(self, "velocity", 0, 0, 2);
	}
	else if ( playerkeys(plr,0, "moveup"))
	{
		changeentityproperty(self, "velocity", 0, 0, -2);
	}
	else
	{
		changeentityproperty(self, "velocity", 0, 0, 0);
	}
}
@end_script
